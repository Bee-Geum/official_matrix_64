	int bid = blockIdx.x;

	if (tid == 0) {
		minDist[bid] = DBL_MAX;
		closestPair[0][bid] = points[bid];

	for (int i = bid + 1; i < numPoints; ++i) {

			closestPair[0][bid] = points[bid];
			closestPair[1][bid] = points[i];
		}
	}
