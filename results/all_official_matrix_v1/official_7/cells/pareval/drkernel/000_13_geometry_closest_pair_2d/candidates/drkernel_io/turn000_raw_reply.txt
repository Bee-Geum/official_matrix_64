    if pid != 0:

        yi = points[2 * i + 1]

            if j < n:
                xj = points[2 * j]
                yj = points[2 * j + 1]

                dy = yj - yi
                dist = tl.sqrt(dx * dx + dy * dy)
                best = tl.minimum(best, dist)
    out[0] = best

    n = pts.shape[0]
    if n <= 1:
        return 0.0 if n == 1 else float('inf')

    px = np.sort(pts, axis=0)
    # Left/right recursive min
    mid = n // 2
    deltaL = _closest_pair_dnc(px[:mid])
    deltaR = _closest_pair_dnc(px[mid:])

    if delta == 0.0:
        return 0.0

import numpy as np
import triton
import triton.language as tl

@triton.jit
def _strip_min_distance(points, n, out):
    # points: shape (2*n,), float32, ordered by y on host before call.
    # Compare each point to up to 7 neighbors; write min to out[0].
    pid = tl.program_id(axis=0)
    if pid != 0:
        return
    inf = 1e38
    best = inf
    for i in range(0, n):
        xi = points[2 * i]
        yi = points[2 * i + 1]
        for d in range(1, 8):
            j = i + d
            if j < n:
                xj = points[2 * j]
                yj = points[2 * j + 1]
                dx = xj - xi
                dy = yj - yi
                dist = tl.sqrt(dx * dx + dy * dy)
                best = tl.minimum(best, dist)
    out[0] = best

def _closest_pair_dnc(pts):
    # pts: (N, 2) float32
    n = pts.shape[0]
    if n <= 1:
        return 0.0 if n == 1 else float('inf')
    # Sort by x
    px = np.sort(pts, axis=0)
    # Left/right recursive min
    mid = n // 2
    deltaL = _closest_pair_dnc(px[:mid])
    deltaR = _closest_pair_dnc(px[mid:])
    delta = min(deltaL, deltaR)
    if delta == 0.0:
        return 0.0
    # Strip: x in [mid_x - delta, mid_x + delta]
    mid_x = px[mid, 0]
    left = px[:mid]
    right = px[mid:]
    # Filter strip
    strip_left = left[np.abs(left[:, 0] - mid_x) <= delta]
    strip_right = right[np.abs(right[:, 0] - mid_x) <= delta]
    strip = np.concatenate([strip_left, strip_right], axis=0)
    if strip.shape[0] == 0:
        return delta
    # Sort strip by y
    strip_y = np.sort(strip, axis=0)
    flat_y = strip_y.reshape(-1)
    # GPU min over neighbors
    out = np.empty(1, dtype=np.float32)
    _strip_min_distance[1](flat_y, strip_y.shape[0], out)
    dstrip = float(out[0])
    return min(delta, dstrip)

    def __init__(self):
