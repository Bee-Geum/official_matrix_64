```python
import torch
import triton
import triton.language as tl

@triton.jit
def _gelu_kernel(x_ptr, y_ptr, n_elements, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask)
    y = 0.5 * x * (1 + tl.math.tanh(tl.math.sqrt(2 / tl.math.pi) * (x + 0.044715 * x**3)))
    tl.store(y_ptr + offsets, y, mask=mask)

def gelu_kernel_impl(input_tensor):
    output_tensor = torch.empty_like(input_tensor)
    n_elements = input_tensor.numel()
    grid = lambda meta: (triton.cdiv(n_elements, meta['BLOCK_SIZE']),)
    _gelu_kernel[grid](input_tensor, output_tensor, n_elements, BLOCK_SIZE=1024)
    return output_tensor

# Example usage
if __name__ == "__main__":
    input_tensor = torch.randn(1024, device='cuda')
    output_tensor = gelu_kernel_impl(input_tensor)
    print(output_tensor)
```
