    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements

    p = tl.load(p_ptr + offsets, mask=mask)
    grad = tl.load(grad_ptr + offsets, mask=mask)
    exp_avg = tl.load(exp_avg_ptr + offsets, mask=mask)
    exp_avg_sq = tl.load(exp_avg_sq_ptr + offsets, mask=mask)

    exp_avg = exp_avg * beta1 + grad * (1 - beta1)
    exp_avg_sq = exp_avg_sq * beta2 + grad * grad * (1 - beta2)

    denom = tl.sqrt(exp_avg_sq) / tl.sqrt(bias_correction2) + eps
    step_size = lr / bias_correction1
    update = exp_avg / denom * step_size
    p_new = p - update

    tl.store(p_ptr + offsets, p_new, mask=mask)
    tl.store(exp_avg_ptr + offsets, exp_avg, mask=mask)
    tl.store(exp_avg_sq_ptr + offsets, exp_avg_sq, mask=mask)

def Adam(params, lr=0.001, betas=(0.9, 0.999), eps=1e-08, weight_decay=0):
    return AdamOptimizer(params, lr=lr, betas=betas, eps=eps, weight_decay=weight_decay)

### Explanation
1. **Custom Optimizer**: The `AdamOptimizer` class initializes optimization parameters and manages the state for each parameter tensor, including moving averages and the current optimization step.
2. **Weight Decay**: Before processing gradients, weight decay is applied to smooth the parameter updates.
3. **Triton Kernel**: The `adam_kernel` function processes each element of the parameter tensors in parallel. It computes the moving averages, applies bias corrections, and updates the parameters using efficient GPU operations.
4. **State Management**: The optimizer maintains states (`exp_avg`, `exp_avg_sq`, and `step`) for each parameter tensor, ensuring correct accumulation of gradient history.
5. **Efficient Execution**: The kernel is launched with a grid configuration that processes elements in blocks, leveraging GPU parallelism for high-performance updates. This approach ensures numerical equivalence to the original PyTorch implementation while optimizing execution speed.</answer>
