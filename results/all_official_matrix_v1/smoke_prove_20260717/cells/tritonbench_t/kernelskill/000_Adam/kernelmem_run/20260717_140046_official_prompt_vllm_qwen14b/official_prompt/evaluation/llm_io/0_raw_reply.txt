# <complete ModelNew code>
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Define the custom CUDA kernel for Adam optimizer step
source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>

__global__ void adam_step_kernel(float* params, float* grads, float* exp_avg, float* exp_avg_sq, float lr, float beta1, float beta2, float eps, float weight_decay, int size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float grad = grads[idx];
        float param = params[idx];
        float avg = exp_avg[idx];
        float avg_sq = exp_avg_sq[idx];

        avg = beta1 * avg + (1 - beta1) * grad;
        avg_sq = beta2 * avg_sq + (1 - beta2) * grad * grad;

        exp_avg[idx] = avg;
        exp_avg_sq[idx] = avg_sq;

        float denom = sqrt(avg_sq) + eps;
        float step = lr * avg / denom;
        param -= step + weight_decay * param;

        params[idx] = param;
    }
}

void adam_step_cuda(torch::Tensor params, torch::Tensor grads, torch::Tensor exp_avg, torch::Tensor exp_avg_sq, float lr, float beta1, float beta2, float eps, float weight_decay) {
    auto size = params.numel();

    const int block_size = 256;
    const int num_blocks = (size + block_size - 1) / block_size;

    auto stream = at::cuda::getDefaultCUDAStream();
    adam_step_kernel<<<num_blocks, block_size, 0, stream>>>(
        params.data_ptr<float>(), grads.data_ptr<float>(), exp_avg.data_ptr<float>(), exp_avg_sq.data_ptr<float>(), lr, beta1, beta2, eps, weight_decay, (int)size
    );
    C10_CUDA_KERNEL_LAUNCH_CHECK();
}
"""

cpp_src = (
    "void adam_step_cuda(torch::Tensor params, torch::Tensor grads, torch::Tensor exp_avg, torch::Tensor exp_avg_sq, float lr, float beta1, float beta2, float eps, float weight_decay);"
)

# Compile the inline CUDA code for Adam optimizer step
adam_step = load_inline(
    name="adam_step",
    cpp_sources=cpp_src,
    cuda_sources=source,
    functions=["adam_step_cuda"],
    verbose=True,
    extra_cflags=[
        "-O3",                          # High optimization
        "-std=c++17",                    # Use C++17 standard
        ],
    extra_ldflags=[""],
    extra_cuda_cflags=[
        "-gencode=arch=compute_90,code=sm_90",  # Correct CUDA target arch
        "--expt-relaxed-constexpr",  # CUDA specific flag
        "-lineinfo",  # Line information for debugging
        ],
)

class ModelNew(nn.Module):
    def __init__(self, lr=0.001, betas=(0.9, 0.999), eps=1e-08, weight_decay=0):
        super().__init__()
        self.lr = lr
        self.betas = betas
        self.eps = eps
        self.weight_decay = weight_decay
        self.param_holder = nn.ParameterList()

    def add_param(self, param):
        self.param_holder.append(param)
        self.register_buffer(f'exp_avg_{len(self.param_holder) - 1}', torch.zeros_like(param))
        self.register_buffer(f'exp_avg_sq_{len(self.param_holder) - 1}', torch.zeros_like(param))

    def zero_grad(self):
        for param in self.param_holder:
            if param.grad is not None:
                param.grad.zero_()

    def step(self):
        for i, param in enumerate(self.param_holder):
            if param.grad is not None:
                exp_avg = getattr(self, f'exp_avg_{i}')
                exp_avg_sq = getattr(self, f'exp_avg_sq_{i}')
                adam_step.adam_step_cuda(param, param.grad, exp_avg, exp_avg_sq, self.lr, self.betas[0], self.betas[1], self.eps, self.weight_decay)

    def forward(self, *args, **kwargs):
        raise NotImplementedError("ModelNew is an optimizer wrapper and does not have a forward pass.")

# Example usage
if __name__ == "__main__":
    model = ModelNew(lr=0.001, betas=(0.9, 0.999), eps=1e-08, weight_decay=0)
    param = torch.randn(2, 2, device='cuda', requires_grad=True)
    model.add_param(param)

    optimizer = torch.optim.Adam([param], lr=0.001, betas=(0.9, 0.999), eps=1e-08, weight_decay=0)

    # Simulate a training step
    output = param * 2
    loss = output.sum()
    loss.backward()

    model.step()
    optimizer.step()

    print(param)
