- Observations:
  - The problem is a classic computational geometry task. The “brute-force” O(n^2) algorithm is straightforward to implement in CUDA but will be slow for large n.
  - A divide-and-conquer algorithm (O(n log n)) is the standard optimal approach, but implementing it efficiently on GPU requires careful handling of recursion, strip processing, and memory access; it’s quite involved.

  - Fallback to a single-block implementation if numPoints <= BLOCK.

  - Set invalid distances (j < i) to a large value to simplify reduction.

  - This is not asymptotically optimal but is simple and parallel. For production, replace with a divide-and-conquer or sweep algorithm.

import triton.language as tl

@triton.jit
def _closest_pair_block(points_ptr,  # *double
                        numPoints: tl.int32,
                        out_ptr,     # *double
                        BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    start = pid * BLOCK
    # If this block is out of range, early exit (not strictly needed if grid is exact).
    if start >= numPoints:
        return

    # Vector of indices for this block
    idx = start + tl.arange(0, BLOCK)
    valid = idx < numPoints

    # Load x, y; invalid lanes get 0
    x = tl.load(points_ptr + 2 * idx, mask=valid, other=0.0)
    y = tl.load(points_ptr + 2 * idx + 1, mask=valid, other=0.0)

    # Initialize min distance to a large value
    min_dist = tl.full((), 1e20, dtype=tl.float32)

    # Compare all pairs within the block: i in [0, BLOCK), j in [i+1, BLOCK)
    # Note: quadratic work; only viable for small BLOCK / small numPoints
    for i in range(BLOCK):
        xi = x[i]
        yi = y[i]
        # Only consider j > i to avoid duplicate work and self-pair
        for j in range(i + 1, BLOCK):
            xj = x[j]
            yj = y[j]
            # Distance
            dx = xi - xj
            dy = yi - yj
            dist = tl.sqrt(dx * dx + dy * dy)
            # Update min
            min_dist = tl.minimum(min_dist, dist)

    # Store result for this block
    tl.store(out_ptr + pid, min_dist)

            ii = chunk_start + local[:, None]   # [BLOCK, S]
            jj = (strip_end - S) + local[None, :]  # [BLOCK, S] -> [BLOCK, S] but equals local[None, :]

            valid_pair = (ii < numPairs) & (jj < numPairs)
            dx = xi_b - xj_b
            dy = yi_b - yj_b
            dist_inter = tl.sqrt(dx * dx + dy * dy)
            big = tl.full(dist_inter.shape, 1e20, dtype=tl.float32)
            dist_inter = tl.where(valid_pair, dist_inter, big)

            min_cols = tl.min(dist_inter, axis=1)  # [BLOCK]
            min_inter = tl.min(min_cols, axis=0)   # scalar
            min_dist = tl.minimum(min_dist, min_inter)

            # Also consider intra-chunk distances for this chunk: i < j within [chunk_start, chunk_end)
            i2 = chunk_start + local[:, None]     # [BLOCK, 1]
            j2 = chunk_start + local[None, :]     # [1, BLOCK]
            valid_i2 = local[:, None] < chunk_size
            valid_j2 = local[None, :] < chunk_size
            ii2 = i2
            jj2 = j2
            valid_pair2 = (ii2 < chunk_end) & (jj2 < chunk_end) & (ii2 < jj2)
            ptr_xi2 = points_ptr + 2 * ii2
            ptr_yi2 = points_ptr + 2 * ii2 + 1
            ptr_xj2 = points_ptr + 2 * jj2
            ptr_yj2 = points_ptr + 2 * jj2 + 1
            xi2_mat = tl.load(ptr_xi2, mask=valid_pair2, other=0.0)
            yi2_mat = tl.load(ptr_yi2, mask=valid_pair2, other=0.0)
            xj2_mat = tl.load(ptr_xj2, mask=valid_pair2, other=0.0)
            yj2_mat = tl.load(ptr_yj2, mask=valid_pair2, other=0.0)
            dx2 = xi2_mat - xj2_mat
            dy2 = yi2_mat - yj2_mat
            dist_in2 = tl.sqrt(dx2 * dx2 + dy2 * dy2)
            big2 = tl.full(dist_in2.shape, 1e20, dtype=tl.float32)
            dist_in2 = tl.where(valid_pair2, dist_in2, big2)
            min_cols2 = tl.min(dist_in2, axis=1)   # [BLOCK]
            min_in2 = tl.min(min_cols2, axis=0)    # scalar
            min_dist = tl.minimum(min_dist, min_in2)

        # Append this chunk's points to the virtual "strip" (no storage needed; just update ends)
        strip_end = chunk_end

    # Store result for this block
    tl.store(out_ptr + pid, min_dist)
