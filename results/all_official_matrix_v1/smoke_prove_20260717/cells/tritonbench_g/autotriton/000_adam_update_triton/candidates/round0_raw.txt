To solve this problem, we need to implement an optimized GPU kernel for the ADAM update step using Triton, ensuring numerical accuracy and performance. The kernel will process each element of the input tensors in parallel, applying the necessary operations to update the parameters and momentum running average.

### Approach
1. **Understanding the ADAM Update**: The ADAM update involves weight decay, difference calculation, update vector generation, sign-based learning rate adjustment, parameter update, and momentum decay. Each of these steps is performed element-wise.
2. **Triton Kernel Design**: The kernel processes elements in blocks, with each thread handling one element. It loads the necessary values, applies the operations, and stores the results back. Masking ensures boundary conditions are handled correctly.
3. **Autotuning**: The kernel uses autotuning to select the optimal block size and number of warps, enhancing performance based on the input size.
4. **Host Function**: The host function ensures tensors are on CUDA, computes the grid size, and launches the kernel with the appropriate parameters.

### Solution Code
```python
import torch
import triton
import triton.language as tl

@triton.autotune(configs=[
    triton.Config({'BLOCK_SIZE': 128}, num_warps=4),
    triton.Config({'BLOCK_SIZE': 1024}, num_warps=8),
], key=['n_elements'], restore_value=['p_ptr', 'exp_avg_ptr'])
@triton.jit
def update_fn_kernel(
    p_ptr,
    grad_ptr,
    exp_avg_ptr,
    lr,
    wd,
    beta1,
    beta2,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements

    offset_p = p_ptr + offsets
    offset_grad = grad_ptr + offsets
    offset_exp_avg = exp_avg_ptr + offsets

    p = tl.load(offset_p, mask=mask)
    grad = tl.load(offset_grad, mask=mask)
    exp_avg = tl.load(offset_exp_avg, mask=mask)

    p = p * (1 - lr * wd)
    diff = exp_avg - grad
    update = diff * beta1 + grad
    can_update = update != 0
    update_sign = tl.where(update > 0, -lr, lr)
    p = p + update_sign * can_update
    exp_avg = diff * beta2 + grad

    tl.store(offset_p, p, mask=mask)
    tl.store(offset_exp_avg, exp_avg, mask=mask)

def update_fn(
    p: torch.Tensor,
    grad: torch.Tensor,
    exp_avg: torch.Tensor,
    lr: float,
    wd: float,
    beta1: float,
    beta2: float
):
    assert all([t.is_cuda for t in (p, grad, exp_avg)]), "All tensors must be on CUDA"
    n_elements = p.numel()
    grid = lambda meta: (triton.cdiv(n_elements, meta['BLOCK_SIZE']),)
    update_fn_kernel[grid](p, grad, exp_avg, lr, wd, beta1, beta2, n_elements)

##################################################################################################################################################

def test_update_fn():
    n_elements = 128
    p1 = torch.randn(n_elements, device='cuda', dtype=torch.float32)
    grad1 = torch.randn(n_elements, device='cuda', dtype=torch.float32)
    exp_avg1 = torch.zeros(n_elements, device='cuda', dtype=torch.float32)

    n_elements = 1024
    p2 = torch.randn(n_elements, device='cuda', dtype=torch.float32)
    grad2 = torch.randn(n_elements, device='cuda', dtype=torch.float32)
    exp_avg2 = torch.zeros(n_elements, device='cuda', dtype=torch.float32)

    lr = 0.01
    wd = 0.01
    beta1 = 0.9
    beta2 = 0.999

    update_fn(p1, grad1, exp_avg1, lr, wd, beta1, beta2)
    update_fn(p2, grad2, exp_avg2, lr, wd, beta1, beta2)

    results = {
        "test_case_1": (p1.clone(), exp_avg1.clone()),
        "test_case_2": (p2.clone(), exp_avg2.clone())
    }
    return results

result_gold = test_update_fn()
```
